CREATE TABLE BOOK(
Isbn CHAR(13) NOT NULL,
Title VARCHAR(100) NOT NULL,
Author VARCHAR(100) NOT NULL,
Publisher VARCHAR(100) NOT NULL,
Year INT NOT NULL,
Price DECIMAL(5,2) NOT NULL,
Category VARCHAR(100) NOT NULL,
PRIMARY KEY (Isbn));

CREATE TABLE INVENTORY(
Isbn CHAR(13) NOT NULL,
Location VARCHAR(100) NOT NULL,
Quantity INT,
PRIMARY KEY (Isbn),
FOREIGN KEY (Isbn) REFERENCES BOOK(Isbn));

CREATE TABLE AUTHOR(
AuthorID INT NOT NULL,
Name VARCHAR(100) NOT NULL,
PRIMARY KEY(AuthorID));

CREATE TABLE PUBLISHER(
Name VARCHAR(100) NOT NULL,
Phone CHAR(11),
Address VARCHAR(100) NOT NULL,
PRIMARY KEY(Name));

CREATE TABLE CUSTOMER(
Email VARCHAR(100) NOT NULL,
Name VARCHAR(100) NOT NULL,
Phone CHAR(11),
Address VARCHAR(100) NOT NULL,
PRIMARY KEY(Email));

CREATE TABLE DISCOUNT(
Isbn CHAR(13) NOT NULL,
Buy_limit INT,
Percentage REAL NOT NULL,
PRIMARY KEY(Isbn),
FOREIGN KEY(Isbn) REFERENCES BOOK(Isbn));

CREATE TABLE ACCOUNT(
Account_no INT NOT NULL,
Email VARCHAR(100) NOT NULL,
Reward_point INT,
Recommendation_list VARCHAR(500),
Membership INT,
PRIMARY KEY(Account_no),
FOREIGN KEY(Email) REFERENCES CUSTOMER(Email));

CREATE TABLE GIFT(
Email VARCHAR(100) NOT NULL,
Type VARCHAR,
PRIMARY KEY(Email),
FOREIGN KEY(Email) REFERENCES CUSTOMER(Email));

CREATE TABLE WRITES(
Isbn CHAR(13) NOT NULL,
Author VARCHAR(100) NOT NULL,
PRIMARY KEY(Isbn,AuthorID),
FOREIGN KEY(Isbn) REFERENCES BOOK(Isbn),
FOREIGN KEY(AuthorID) REFERENCES AUTHOR(AuthorID));

CREATE TABLE PUBLISHES(
Isbn CHAR(13) NOT NULL,
Publisher VARCHAR(100) NOT NULL,
PRIMARY KEY(Isbn,Publisher),
FOREIGN KEY(Isbn) REFERENCES BOOK(Isbn),
FOREIGN KEY(Publisher) REFERENCES PUBLISHER(Name));

CREATE TABLE PURCHASE(
Transaction_id INT NOT NULL,
Customer VARCHAR(100) NOT NULL,
Book CHAR(13) NOT NULL,
Actual_Price DECIMAL(5,2) NOT NULL,
Date DATE,
Card_type VARCHAR(100),
PRIMARY KEY(Transaction_id),
FOREIGN KEY(Customer) REFERENCES CUSTOMER(Email),
FOREIGN KEY(Book) REFERENCES BOOK(Isbn));
